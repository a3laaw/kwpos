   307→  function addToCart(p: Product) {
   308→    const inCartQty = inCart.get(p.id) || 0
   309→    if (inCartQty >= p.quantity) {
   310→      toast.error("الكمية غير متوفرة", {
   311→        description: `المتاح من ${p.name}: ${fmt.number(p.quantity)} ${p.unit}`,
   312→      })
   313→      return
   314→    }
   315→    const existing = cart.find((it) => it.product.id === p.id)
   316→    if (!existing) {
   317→      // New item — auto-advance to the page it will appear on
   318→      const newIndex = cart.length // index of the new item
   319→      const newPage = Math.floor(newIndex / ITEMS_PER_CART_PAGE)
   320→      setCartPage(newPage)
   321→    }
   322→    setCart((c) => {
   323→      const ex = c.find((it) => it.product.id === p.id)
   324→      if (ex) {
   325→        return c.map((it) =>
   326→          it.product.id === p.id ? { ...it, quantity: it.quantity + 1 } : it
   327→        )
   328→      }
   329→      return [...c, { product: p, quantity: 1 }]
   330→    })
   331→  }

   333→  function changeQty(productId: string, delta: number) {
   334→    setCart((c) =>
   335→      c
   336→        .map((it) => {
   337→          if (it.product.id !== productId) return it
   338→          const next = it.quantity + delta
   339→          if (next > it.product.quantity) {
   340→            toast.error("الكمية تجاوزت المخزون المتاح")
   341→            return it
   342→          }
   343→          return { ...it, quantity: next }
   344→        })
   345→        .filter((it) => it.quantity > 0)
   346→    )
   347→  }
   348→
   349→  function setQty(productId: string, qty: number) {
   350→    const p = cart.find((it) => it.product.id === productId)?.product
   351→    if (!p) return
   352→    if (qty > p.quantity) {
   353→      toast.error("الكمية تجاوزت المخزون المتاح")
   354→      return
   355→    }
   356→    setCart((c) =>
   357→      qty <= 0
   358→        ? c.filter((it) => it.product.id !== productId)
   359→        : c.map((it) => (it.product.id === productId ? { ...it, quantity: qty } : it))
   360→    )
   361→  }
   362→
   363→  function removeItem(productId: string) {
   364→    setCart((c) => c.filter((it) => it.product.id !== productId))
   365→  }

   367→  function clearCart() {
   368→    setCart([])
   369→    setDiscount("0")
   370→    setCustomerName("")
   371→    setCustomerPhone("")
   372→    setDeliveryEnabled(false)
   373→    setDriverName("")
   374→    setDeliveryFee("")
   375→  }
   376→
   377→  const subtotal = cart.reduce((acc, it) => acc + priceFor(it.product) * it.quantity, 0)
   378→  const discountVal = Math.max(0, Math.min(Number(discount) || 0, subtotal))
   379→  const deliveryFeeVal = deliveryEnabled ? Math.max(0, Number(deliveryFee) || 0) : 0
   380→  const afterDiscount = Math.max(0, subtotal - discountVal)
   381→  const taxVal = +(afterDiscount * ((Number(taxRate) || 0) / 100)).toFixed(2)
   382→  const total = +(afterDiscount + taxVal + deliveryFeeVal).toFixed(2)
   383→  const itemCount = cart.reduce((a, b) => a + b.quantity, 0)
   384→
   385→  // Sale confirmation summary shown in the warning dialog before committing.
   386→  const confirmSummary: SaleConfirmSummary | null = React.useMemo(
   387→    () => ({
   388→      itemCount,
   389→      subtotal,
   390→      discount: discountVal,
   391→      taxAmount: taxVal,
   392→      taxRate: Number(taxRate) || 0,
   393→      deliveryFee: deliveryFeeVal,
   394→      driverName: deliveryEnabled ? driverName.trim() || null : null,
   395→      total,
   396→      paymentMethod,
   397→      customerName: customerName.trim() || null,
   398→    }),
   399→    [itemCount, subtotal, discountVal, taxVal, taxRate, deliveryFeeVal, deliveryEnabled, driverName, total, paymentMethod, customerName]
   400→  )

   402→  // Clicking "إتمام البيع" does NOT execute the sale directly — it opens the

   405→  function handleCheckout() {

   407→      toast.error("السلة فارغة")
   408→      return
   409→    }
   410→    setConfirmOpen(true)
   411→  }

   413→  async function doConfirmSale() {
   414→    if (cart.length === 0) return
   415→    try {
   416→      const sale = await createMut.mutateAsync({
   417→        customerName: customerName.trim() || undefined,
   418→        customerPhone: customerPhone.trim() || undefined,
   419→        items: cart.map((it) => ({
   420→          productId: it.product.id,
   421→          quantity: it.quantity,
   422→          unitPrice: priceFor(it.product),
   423→        })),
   424→        taxRate: Number(taxRate) || 0,
   425→        discount: discountVal,
   426→        paymentMethod,
   427→        deliveryFee: deliveryFeeVal,
   428→        driverName: deliveryEnabled ? driverName.trim() || undefined : undefined,
   429→      })
   430→      toast.success("تمت عملية البيع بنجاح", {
   431→        description: `رقم الفاتورة: ${sale.invoiceNo}`,
   432→      })
   433→      setLastSale(sale)
   434→      clearCart()
   435→    } catch (err: any) {
   436→      // Stale session (e.g. after re-seed) — auto-logout so user can re-login.
   437→      if (err?.message === "session-expired") {
   438→        toast.error("انتهت الجلسة", {
   439→          description: "يرجى تسجيل الدخول مرة أخرى.",
   440→        })
   441→        setTimeout(() => {
   442→          signOut({ redirect: false }).then(() => window.location.reload())
   443→        }, 1500)
   444→        return
   445→      }
   446→      // Handle stock-insufficient errors specifically
   447→      if (err?.message?.startsWith("stock-insufficient")) {
   448→        const parts = err.message.split(":")
   449→        const name = parts[2] || "منتج"
   450→        toast.error("كمية غير كافية في المخزون", {
   451→          description: `الرصيد المتاح من ${name} لا يكفي. تم تحديث الكميات.`,
   452→        })
   453→      } else {
   454→        toast.error("فشل إتمام البيع", { description: err?.message })
   455→      }
   456→    } finally {
   457→      setConfirmOpen(false)
   458→    }
   459→  }
   460→
   461→  return (
   462→    <div className="space-y-5">
   463→      <PageHeader
   464→        title="نقاط البيع (POS)"
   465→        description="أنشئ فاتورة مبيعات جديدة — ابحث عن المنتجات وأضفها للسلة، احسب الإجمالي والضريبة والخصم تلقائياً."
   466→        icon={<Calculator className="h-5 w-5" />}
   467→      />
   468→
   469→      <div className="grid grid-cols-1 lg:grid-cols-5 gap-5">
   470→        {/* Product picker */}
   471→        <div className="lg:col-span-3 space-y-4">
   472→          <div className="relative">
   473→            <Search className="absolute right-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
   474→            <Input
   475→              value={q}
   476→              onChange={(e) => setQ(e.target.value)}
   477→              placeholder="ابحث عن منتج بالاسم أو الباركود..."
   478→              className="pr-9 h-11 text-base"
   479→              autoFocus
   480→            />
   481→          </div>
   482→
   483→          {/* Category filter cards */}
   484→          {categories.length > 0 ? (
   485→            <div className="flex gap-2 overflow-x-auto scrollbar-thin pb-1">
   486→              <button
   487→                onClick={() => setCategoryId("")}
   488→                className={cn(
   489→                  "shrink-0 flex items-center gap-1.5 rounded-lg px-3 py-2 text-sm font-medium transition-all border",
   490→                  !categoryId
   491→                    ? "bg-primary text-primary-foreground border-primary shadow-sm"
   492→                    : "bg-card border-border/70 hover:border-primary/40 text-foreground"
   493→                )}
   494→              >
   495→                <LayoutGrid className="h-4 w-4" />
   496→                الكل
   497→              </button>
   498→              {categories.map((c) => {
   499→                const active = categoryId === c.id
   500→                const Icon = CATEGORY_ICONS[c.name] || Tag
   501→                return (
   502→                  <button
   503→                    key={c.id}
   504→                    onClick={() => setCategoryId(active ? "" : c.id)}
   505→                    className={cn(
   506→                      "shrink-0 flex items-center gap-1.5 rounded-lg px-3 py-2 text-sm font-medium transition-all border",
   507→                      active
   508→                        ? "bg-primary text-primary-foreground border-primary shadow-sm"
   509→                        : "bg-card border-border/70 hover:border-primary/40 text-foreground"
   510→                    )}
   511→                  >
   512→                    {c.imageUrl ? (
   513→                      <img src={c.imageUrl} alt="" className="h-4 w-4 rounded object-cover" />
   514→                    ) : (
   515→                      <Icon className="h-4 w-4" />
   516→                    )}
   517→                    {c.name}
   518→                  </button>
   519→                )
   520→              })}
   521→            </div>
   522→          ) : null}
   523→
   524→          {isLoading ? (
   525→            <div className="grid grid-cols-2 sm:grid-cols-3 gap-3">
   526→              {Array.from({ length: 6 }).map((_, i) => (
   527→                <div key={i} className="h-28 rounded-xl bg-muted/50 animate-pulse" />
