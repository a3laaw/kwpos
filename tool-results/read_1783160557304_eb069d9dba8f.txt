    27‚Üí    27‚Üíasync function jget<T>(url: string): Promise<T> {
    28‚Üí    28‚Üí  const res = await fetch(url, { headers: { Accept: "application/json" } })
    29‚Üí    29‚Üí  if (!res.ok) {
    30‚Üí    30‚Üí    const e = await res.json().catch(() => ({}))
    31‚Üí    31‚Üí    throw new Error((e as any)?.error || `request-failed:${res.status}`)
    32‚Üí    32‚Üí  }
    33‚Üí    33‚Üí  return res.json() as Promise<T>
    34‚Üí    34‚Üí}

    36‚Üí    36‚Üíasync function jsend<T>(url: string, method: string, body?: unknown): Promise<T> {
    37‚Üí    37‚Üí  const res = await fetch(url, {
    38‚Üí    38‚Üí    method,
    39‚Üí    39‚Üí    headers: { "Content-Type": "application/json", Accept: "application/json" },
    40‚Üí    40‚Üí    body: body ? JSON.stringify(body) : undefined,
    41‚Üí    41‚Üí  })
    42‚Üí    42‚Üí  const data = await res.json().catch(() => ({}))
    43‚Üí    43‚Üí  if (!res.ok) {
    44‚Üí    44‚Üí    throw new Error((data as any)?.error || `request-failed:${res.status}`)
    45‚Üí    45‚Üí  }
    46‚Üí    46‚Üí  return data as T
    47‚Üí    47‚Üí}
